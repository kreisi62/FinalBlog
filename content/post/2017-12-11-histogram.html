library(Lahman)
library(sqldf)
library(ggplot2)
# extracting the data-----------------------------------------

query<-"SELECT weight
FROM Master"

result<-sqldf(query)

# visualize the data-----------------------------------------

ggplot()+
  geom_histogram(data=result,aes(x=weight),color="blue",fill="white",bins=50)+
  ggtitle("Body-Weight Distribution for Baseball Players")
